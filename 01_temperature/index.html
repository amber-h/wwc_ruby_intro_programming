Topics:

Hints

To convert from fahrenheit to celsius, subtract 32 and multiply by 5/9

Remember, in fixnum math, there are no fractions. So if you are using fixnums...

1 / 2 => 0

In float math, there are fractions. So...

1.0 / 2.0 => 0.5

Tests

temperature_spec.rb 
require "temperature"

describe "temperature conversion functions" do

  describe "#fahrenheit_to_celsius" do

    it "converts freezing temperature" do
      fahrenheit_to_celsius(32).should == 0
    end

    it "converts boiling temperature" do
      fahrenheit_to_celsius(212).should == 100
    end

    it "converts body temperature" do
      fahrenheit_to_celsius(98.6).should == 37
    end

    it "converts a temperature and rounds to 1 decimal place" do
      fahrenheit_to_celsius(68).should == 20
    end

  end

  describe "#celsius_to_fahrenheit" do

    it "converts freezing temperature" do
      celsius_to_fahrenheit(0).should == 32
    end

    it "converts boiling temperature" do
      celsius_to_fahrenheit(100).should == 212
    end

    it "converts a temperature and rounds to 1 decimal place" do
      celsius_to_fahrenheit(20).should == 68
    end

  end

end
